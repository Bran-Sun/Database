<program>  := <stmt> <stmtList>
            | %empty
<stmtList> := <stmt> <stmtList>
            | %empty

<stmt>  := <showStmt> ’;’ //done
           | <createStmt> ';'   //done
         | <useStmt>’;’     //done
         | <tbStmt>’;’      //done
         | <idxStmt>’;’     //done

<showStmt> := SHOW DATABASES    //done
            | SHOW TABLES   //done

<createStmt> := CREATE DATABASE <dbName>    //done
            | CREATE TABLE <tbName> '(' <fieldList> ')' //done
            | CREATE INDEX <tbName> '(' IDENTIFIER(colName) ')' //done

<dropStmt> := DROP DATABASE <dbName>    //done
            | DROP TABLE <tbName>   //done
            | DROP INDEX <tbName> '('<colName>')'   //done


<useStmt>  := USE <dbName>

<tbStmt>  := DESC <tbName>
           | INSERT INTO <tbName> VALUES <valueLists>
           | DELETE FROM <tbName> WHERE <whereClause>
           | UPDATE <tbName> SET <setClause> WHERE <whereClause>
           | SELECT <selector> FROM <tableList> WHERE <whereClause>

<fieldList>  := <field> <fieldNext> {'identifier', primary foreign} //done
            | empty  {')'}  //done

<fieldNext> := ',' <field> <filedNext>  //done
            | empty {')'}   //done

<field>  := <colName> <type> <nullReciever>     //done
      | PRIMARY KEY ’(’<column>’)’   //done
      | FOREIGN KEY ’(’<colName>’)’ REFERENCES <tbName>’(’<colName>’)’  //done

<nullReceiver> := NOT NULL      //done
            | empty {')', ','}  //done

<type>  := INT '(' <VALUE_INT> ')'  //done
         | VARCHAR '(' <VALUE_INT> ')'  //done
         | DATE     //done
         | FLOAT    //done

<valueLists>  := ’(’<valueList>’)’
| <valueLists>’, ’ ’ (’<valueList>’) ’

<valueList>  := <value>
| <valueList>’, ’<value>

<value>  := VALUE_INT
          | VALUE_STRING
          | NULL

<whereClause>  := <col><op><expr>
                | <col> IS [NOT] NULL
                | <whereClause> AND <whereClause>

<col>  := [<tbName>’.’]<colName>

<op>  := ‘=’ | ‘<>’ | ‘<=’ | ‘>=’ | ‘<’ | ‘>’

<expr>  := <value>
         | <col>

<setClause>  := <colName>’=’<value>
| <setClause>’, ’<colName>’=’<value>

<selector>  := * | <col>[,<col>]*

<tableList>  := <tbName>
| <tableList>’,’<tbName>

<columnList>  := <colName>
            | <columnList>’,’<colName>

<dbName>  := IDENTIFIER

<tbName>  := IDENTIFIER

<colName> := IDENTIFIER
